---
title: Falcon Deployment
---

Run the `build` & `start` commands with the `:prod` flag from both the `server` and `client` to build and start your app.

import Tabs from '@theme/Tabs';
import TabItem from '@theme/TabItem';

<Tabs
  defaultValue="npm"
  values={[
    { label: 'NPM', value: 'npm', },
    { label: 'Yarn', value: 'yarn', }
  ]
}>
<TabItem value="npm">

```bash
npm build:prod
npm start:prod
```

</TabItem>
<TabItem value="yarn">

```bash
yarn build:prod
yarn start:prod
```

</TabItem>
</Tabs>
